```
function fibonacci(n) {
  if(n < 2){return 1};
  return fibonacci(n - 1) + fibonacci(n - 2)
}
```

2.循环

```
function fibonacci(n) {
  let num1 = 1;
  let num2 = 1;
  let sum = num2;
  for(let n = 0; n < n.length; n ++) {
    sum = num1 + num2;
    num1 = num2;
    num2 = sum;
  }
  return sum;
}
```
